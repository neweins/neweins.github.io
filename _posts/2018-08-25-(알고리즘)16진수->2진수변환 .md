void convertToBit()
{
    for(int i=0; i< strIn.length(); ++i){
        switch(strIn[i]){
            case '0' : strbit.append("0000"); break;
            case '1' : strbit.append("0001"); break;
            case '2' : strbit.append("0010"); break;
            case '3' : strbit.append("0011"); break;
            case '4' : strbit.append("0100"); break;
            case '5' : strbit.append("0101"); break;
            case '6' : strbit.append("0110"); break;
            case '7' : strbit.append("0111"); break;
            case '8' : strbit.append("1000"); break;
            case '9' : strbit.append("1001"); break;
            case 'A' : strbit.append("1010"); break;
            case 'B' : strbit.append("1011"); break;
            case 'C' : strbit.append("1100"); break;
            case 'D' : strbit.append("1101"); break;
            case 'E' : strbit.append("1110"); break;
            case 'F' : strbit.append("1111"); break;
                
        }
    }
}


int main()
{
    scanf("%X", &input);
    sprintf(sInput,"%X", input);
    strIn.append(sInput);
    
    //16진수를 2진수 32bit로 변환해야 함.
    convertToBit();
    cout << strbit <<endl;
    
    int index=0;
    for(int i=32-strbit.length(); i <32; ++i){
         strFullbit[i] = strbit[index++];
