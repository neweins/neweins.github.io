# (알고리즘)16진수(4바이트)->2진수(32비트)변환

## c++ 구현

~~~c

 16진수 4byte 입력 받아 2진수 변환 후 비트 재배치하고 다시 16진수 변환 후 출력

#include <string>
#include <iostream>

using namespace std;

void convertHexToBit(string& strHexIn, string& strBitOut)
{
    strBitOut.assign("00000000000000000000000000000000");
    string strTmp;
    for(int i=0; i< strHexIn.length(); ++i){
        switch(strHexIn[i]){
            case '0' : strTmp.append("0000"); break;
            case '1' : strTmp.append("0001"); break;
            case '2' : strTmp.append("0010"); break;
            case '3' : strTmp.append("0011"); break;
            case '4' : strTmp.append("0100"); break;
            case '5' : strTmp.append("0101"); break;
            case '6' : strTmp.append("0110"); break;
            case '7' : strTmp.append("0111"); break;
            case '8' : strTmp.append("1000"); break;
            case '9' : strTmp.append("1001"); break;
            case 'A' : strTmp.append("1010"); break;
            case 'B' : strTmp.append("1011"); break;
            case 'C' : strTmp.append("1100"); break;
            case 'D' : strTmp.append("1101"); break;
            case 'E' : strTmp.append("1110"); break;
            case 'F' : strTmp.append("1111"); break;
            case 'a' : strTmp.append("1010"); break;
            case 'b' : strTmp.append("1011"); break;
            case 'c' : strTmp.append("1100"); break;
            case 'd' : strTmp.append("1101"); break;
            case 'e' : strTmp.append("1110"); break;
            case 'f' : strTmp.append("1111"); break;
                
        }
    }
    
    
    
    int index=0;
    for(int i=32-(int)strTmp.length(); i <32; ++i){
        strBitOut[i] = strTmp[index];
        index++;
    }
}


void swap(string& strdest, size_t srcIdx,size_t srclen, size_t destIdx, size_t destlen)
{
    string strTmp;
    strTmp.append(&strdest[srcIdx], srclen);
    strdest.replace(strdest.begin()+ srcIdx, strdest.begin()+srcIdx+srclen, strdest.begin()+destIdx,strdest.begin()+destIdx+destlen);
    strdest.replace(strdest.begin()+destIdx,strdest.begin()+destIdx+destlen,strTmp);
}


int main()
{
    string strHexIn;
    string strBitOut;

    
    cin >> strHexIn;
   
    //16진수를 2진수 32bit로 변환해야 함.
    convertHexToBit(strHexIn, strBitOut);
    swap(strBitOut,0, 4, 28, 4);
    for(int i=0; i< strBitOut.length(); ++i)
        cout <<strBitOut[i];
    cout << endl;
    swap(strBitOut,0, 4, 28, 4);
    for(int i=0; i< strBitOut.length(); ++i)
        cout <<strBitOut[i];
    cout << endl;

    return 0;
}
~~~
